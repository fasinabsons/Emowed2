  expires_at TIMESTAMP DEFAULT (NOW() + INTERVAL '48 hours')

  engaged_date DATE,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Weddings Table

  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),

  venue TEXT,
  mode TEXT DEFAULT 'combined', -- combined, separate
  budget_limit DECIMAL,
