**Progress Calculation:**
```typescript
const calculateWeddingProgress = (wedding) => {
  const milestones = [
    { complete: !!wedding.id, weight: 10 },
    { complete: wedding.events?.length > 0, weight: 10 },
    { complete: wedding.guests?.length >= 50, weight: 20 },
    { complete: wedding.vendors?.length >= 3, weight: 20 },
    { complete: !!wedding.wedding_card_id, weight: 15 },
    { complete: wedding.media?.length > 0, weight: 10 },
    { complete: wedding.games?.length > 0, weight: 10 },
    { complete: wedding.rsvp_rate >= 0.8, weight: 5 }
  ];
  
  const completed = milestones
    .filter(m => m.complete)
    .reduce((sum, m) => sum + m.weight, 0);
  
  return completed;
};
```

## REAL-TIME UPDATES

```typescript
useEffect(() => {
  // Subscribe to wedding changes
  const weddingChannel = supabase
    .channel(`wedding:${weddingId}`)
    .on(
      'postgres_changes',
      {
        event: '*',
        schema: 'public',
        table: 'guests',
        filter: `wedding_id=eq.${weddingId}`
      },
      (payload) => {
        if (payload.eventType === 'INSERT') {
          toast.success(`${payload.new.full_name} was invited!`);
          fetchGuestCount();
        }
      }
    )
    .on(
      'postgres_changes',
      {
        event: '*',
        schema: 'public',
        table: 'rsvps',
        filter: `wedding_id=eq.${weddingId}`
      },
      (payload) => {
        if (payload.new.status === 'attending') {
          toast.success('New RSVP confirmed!');
          fetchRSVPCount();
        }
      }
    )
    .subscribe();
  
  return () => {
    weddingChannel.unsubscribe();
  };
}, [weddingId]);
```
