## DATABASE SCHEMA

### game_types table
```sql
CREATE TABLE game_types (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  game_code TEXT UNIQUE NOT NULL,
  game_name TEXT NOT NULL,
  game_category TEXT CHECK (game_category IN ('quiz', 'prediction', 'photo_challenge', 'trivia', 'voting', 'custom')),
  description TEXT,
  rules TEXT,
  min_players INTEGER DEFAULT 1,
  max_players INTEGER,
  duration_minutes INTEGER,
  is_team_based BOOLEAN DEFAULT FALSE,
  scoring_method TEXT CHECK (scoring_method IN ('points', 'time', 'accuracy', 'votes')),
  is_active BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_game_types_category ON game_types(game_category);
CREATE INDEX idx_game_types_active ON game_types(is_active);
```

## FUNCTIONS & TRIGGERS

### Update participant score
```sql
CREATE OR REPLACE FUNCTION update_participant_score()
RETURNS TRIGGER AS $$
BEGIN
  UPDATE game_participants
  SET total_score = total_score + NEW.points_earned,
      games_played = games_played + 1,
      last_played_at = NOW()
  WHERE id = NEW.participant_id;
  
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER after_game_response
AFTER INSERT ON game_responses
FOR EACH ROW
EXECUTE FUNCTION update_participant_score();
```

### Update side competition scores
```sql
CREATE OR REPLACE FUNCTION update_side_competition()
RETURNS TRIGGER AS $$
DECLARE
  v_wedding_id UUID;
  v_side TEXT;
  v_points INTEGER;
BEGIN
  SELECT wg.wedding_id, gp.side, SUM(gr.points_earned)
  INTO v_wedding_id, v_side, v_points
  FROM game_participants gp
  JOIN wedding_games wg ON gp.wedding_game_id = wg.id
  JOIN game_responses gr ON gr.participant_id = gp.id
  WHERE gp.id = NEW.participant_id
  GROUP BY wg.wedding_id, gp.side;

  IF v_side = 'groom' THEN
    UPDATE wedding_side_competition
    SET groom_side_score = groom_side_score + NEW.points_earned
    WHERE wedding_id = v_wedding_id;
  ELSIF v_side = 'bride' THEN
    UPDATE wedding_side_competition
    SET bride_side_score = bride_side_score + NEW.points_earned
    WHERE wedding_id = v_wedding_id;
  END IF;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER after_game_score_update
AFTER INSERT ON game_responses
FOR EACH ROW
EXECUTE FUNCTION update_side_competition();
```

### Calculate leaderboard ranks
```sql
CREATE OR REPLACE FUNCTION calculate_singles_ranks()
RETURNS VOID AS $$
BEGIN
  WITH ranked_users AS (
    SELECT 
      id,
      ROW_NUMBER() OVER (ORDER BY overall_score DESC) as overall_rank,
      ROW_NUMBER() OVER (ORDER BY personal_growth_score DESC) as personal_rank,
      ROW_NUMBER() OVER (ORDER BY career_achievement_score DESC) as career_rank
    FROM singles_leaderboard
  )
  UPDATE singles_leaderboard sl
  SET 
    rank_overall = ru.overall_rank,
    rank_personal = ru.personal_rank,
    rank_career = ru.career_rank
  FROM ranked_users ru
  WHERE sl.id = ru.id;
END;
$$ LANGUAGE plpgsql;
```

### Update singles score on activity
```sql
CREATE OR REPLACE FUNCTION update_singles_score_on_activity()
RETURNS TRIGGER AS $$
BEGIN
  IF NEW.verified = TRUE THEN
    IF NEW.category = 'personal' OR NEW.category = 'both' THEN
      UPDATE singles_leaderboard
      SET personal_growth_score = personal_growth_score + NEW.points_earned,
          overall_score = overall_score + NEW.points_earned
      WHERE user_id = NEW.user_id;
    END IF;
    
    IF NEW.category = 'career' OR NEW.category = 'both' THEN
      UPDATE singles_leaderboard
      SET career_achievement_score = career_achievement_score + NEW.points_earned,
          overall_score = overall_score + NEW.points_earned
      WHERE user_id = NEW.user_id;
    END IF;
  END IF;
  
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER after_activity_verification
AFTER UPDATE ON singles_activities
FOR EACH ROW
WHEN (NEW.verified = TRUE AND OLD.verified = FALSE)
EXECUTE FUNCTION update_singles_score_on_activity();
```

### Calculate gift commission
```sql
CREATE OR REPLACE FUNCTION calculate_gift_commission()
RETURNS TRIGGER AS $$
DECLARE
  v_commission_rate DECIMAL(5,2);
  v_commission_amount DECIMAL(10,2);
BEGIN
  IF NEW.product_id IS NOT NULL THEN
    SELECT commission_rate INTO v_commission_rate
    FROM gift_products
    WHERE id = NEW.product_id;
    
    v_commission_amount := NEW.gift_amount * v_commission_rate / 100;
    
    NEW.commission_amount := v_commission_amount;
  END IF;
  
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER before_gift_insert
BEFORE INSERT ON guest_gifts
FOR EACH ROW
EXECUTE FUNCTION calculate_gift_commission();
```

### Update photo challenge votes
```sql
CREATE OR REPLACE FUNCTION update_photo_challenge_votes()
RETURNS TRIGGER AS $$
BEGIN
  IF TG_OP = 'INSERT' THEN
    UPDATE photo_challenge_submissions
    SET votes_count = votes_count + 1
    WHERE id = NEW.submission_id;
  ELSIF TG_OP = 'DELETE' THEN
    UPDATE photo_challenge_submissions
    SET votes_count = votes_count - 1
    WHERE id = OLD.submission_id;
  END IF;
  RETURN NULL;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER after_photo_vote
AFTER INSERT OR DELETE ON photo_challenge_votes
FOR EACH ROW
EXECUTE FUNCTION update_photo_challenge_votes();
```

## USER FLOWS

### Flow 1: Couple Creates Wedding Game

**Trigger:** Couple wants to add games for guest engagement

**Steps:**
1. Navigate to Games section
2. Browse game templates:
   - Quiz: "How well do you know the couple?"
   - Prediction: "Guess the couple's future"
   - Photo Challenge: "Best wedding moment"
   - Trivia: "Wedding traditions around the world"
   - Custom: Create your own
3. Select game template
4. Configure game:
   - Title (can customize)
   - Description
   - Enable/disable for guests
   - Team-based or individual
   - Side-specific or combined
   - Start/end times (optional)
5. Add questions (if quiz/trivia):
   - Question text
   - Answer options
   - Correct answer
   - Points per question
   - Time limit
6. Preview game
7. Publish game
8. Guests receive notification
9. Game appears in wedding app

## BUSINESS RULES

### Games
- Couple can create unlimited games
- Each game can have 5-50 questions
- Games can be time-limited or always available
- Side-based competition optional
- Prizes optional but recommended
- Winners announced automatically at end time
- Guests can replay games (lowest score kept)

### Photo Challenges
- Max 5 submissions per guest
- Voting starts after submission deadline
- Each guest gets 5 votes
- Cannot vote for own submissions
- Winner = most votes
- Ties broken by submission timestamp

### Leaderboards
- **Singles:**
  - Personal: Fitness, books, skills, volunteer
  - Career: Certifications, projects, promotions
  - Overall: Combined score
  - Verification required for points
  - Admins can verify or auto-verify with proof
  
- **Couples:**
  - Milestones auto-award points
  - Privacy settings respected
  - Points range: 50-500 per milestone
  - Badges at 5, 10, 25, 50 milestones

### Gifting
- **Commission Structure:**
  - Physical products: 5% commission
  - Gift cards: 3% commission
  - Cash gifts: No commission
  
- **Delivery:**
  - Physical: 5-15 days
  - Digital: Instant
  - Gift cards: Instant
  
- **Returns:**
  - 7-day return window
  - Refund to original payment method
  - Cash gifts non-refundable

### Side Competition
- Points accumulated from all games
- Real-time leaderboard
- Winner announced at reception end
- Optional prizes for winning side
- Individual winners also recognized
